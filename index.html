    <div class="card">

        <h1>🎧 華語聽音辨位</h1>

        <div class="subtitle">依據教育部拼音系統設計</div>

       

        <select id="lessonSelect" onchange="changeLesson()">

            <option value="b_p_m">雙唇音對比練習 (b, p, m)</option>

            <option value="d_t_n_l">舌尖音對比練習 (d, t, n, l)</option>

        </select>

 

        <p style="color:#486581; font-weight:bold;">請點擊播放，選出聽到的拼音：</p>

        <button class="btn" onclick="playCurrentWord()" style="background-color: #ff3860;">🔊 播放真人發音</button>

       

        <div id="options-container" style="margin-top: 25px; display: flex; flex-wrap: wrap; justify-content: center;"></div>

        <div id="feedback"></div>

        <button class="btn" onclick="nextQuestion()" style="background-color: #829ab1; margin-top: 15px;">下一題 ➡</button>

    </div>

 

    <div class="card recorder-zone">

        <h1>🎤 錄音自我檢測</h1>

        <p class="subtitle">聽完標準音後，點擊允許麥克風並試著自己唸唸看</p>

        <button class="btn" id="startRec" style="background-color: #209cee;">🔴 開始錄音</button>

        <button class="btn" id="stopRec" style="background-color: #829ab1;" disabled>⏹️ 停止並回放</button>

        <div id="audioPlayback" style="margin-top: 15px;"></div>

    </div>

 

    <script>

        // 💡 這裡是你未來要維護的題庫！

        // 你可以把 audioUrl 換成你自己錄製的 MP3 網址，或是教育部字典的音檔連結

        const lessons = {

            b_p_m: [

                { word: "爸", pinyin: "bà", alternatives: ["pà", "mā", "dà"], audioUrl: "https://d1vxppiqogw2dt.cloudfront.net/dict/audio/ba4.mp3" },

                { word: "怕", pinyin: "pà", alternatives: ["bà", "tā", "kà"], audioUrl: "https://d1vxppiqogw2dt.cloudfront.net/dict/audio/pa4.mp3" }

            ],

            d_t_n_l: [

                { word: "帶", pinyin: "dài", alternatives: ["tài", "nài", "lài"], audioUrl: "https://d1vxppiqogw2dt.cloudfront.net/dict/audio/dai4.mp3" }

            ]

        };

 

        let currentLesson = "b_p_m";

        let currentIndex = 0;

        let currentAudio = null;

 

        function changeLesson() {

            currentLesson = document.getElementById('lessonSelect').value;

            currentIndex = 0;

            loadQuestion();

        }

 

        // 播放真實的 MP3 音檔

        function playCurrentWord() {

            const current = lessons[currentLesson][currentIndex];

            if (currentAudio) { currentAudio.pause(); }

            currentAudio = new Audio(current.audioUrl);

            currentAudio.play();

        }

 

        function loadQuestion() {

            const container = document.getElementById('options-container');

            const feedback = document.getElementById('feedback');

            container.innerHTML = '';

            feedback.innerText = '';

           

            const current = lessons[currentLesson][currentIndex];

            const allOptions = [current.pinyin, ...current.alternatives].sort(() => Math.random() - 0.5);

 

            allOptions.forEach(opt => {

                const btn = document.createElement('button');

                btn.className = 'btn btn-pinyin';

                btn.innerHTML = `${opt}`;

                btn.onclick = () => checkAnswer(opt, btn);

                container.appendChild(btn);

            });

        }

 

        function checkAnswer(selected, btn) {

            const current = lessons[currentLesson][currentIndex];

            const buttons = document.querySelectorAll('.btn-pinyin');

            buttons.forEach(b => b.disabled = true);

 

            if (selected === current.pinyin) {

                btn.classList.add('correct');

                document.getElementById('feedback').innerHTML = `✅ 答對了！<br><span style="font-size:1.5rem; color:#102a43;">${current.word} (${current.pinyin})</span>`;

                document.getElementById('feedback').style.color = "#48c774";

            } else {

                btn.classList.add('wrong');

                document.getElementById('feedback').innerHTML = `❌ 不對喔，正確是 <br><span style="font-size:1.5rem; color:#102a43;">${current.word} (${current.pinyin})</span>`;

                document.getElementById('feedback').style.color = "#f14668";

            }

        }

 

        function nextQuestion() {

            currentIndex = (currentIndex + 1) % lessons[currentLesson].length;

            loadQuestion();

        }

 

        // 錄音功能：移出 GAS 沙盒後，這段代碼就能正常喚起麥克風了

        let mediaRecorder;

        let audioChunks = [];

 

        document.getElementById('startRec').onclick = async () => {

            try {

                const stream = await navigator.mediaDevices.getUserMedia({ audio: true });

                mediaRecorder = new MediaRecorder(stream);

                mediaRecorder.start();

                audioChunks = [];

                document.getElementById('startRec').disabled = true;

                document.getElementById('stopRec').disabled = false;

                mediaRecorder.ondataavailable = e => audioChunks.push(e.data);

            } catch (err) {

                alert("無法存取麥克風，請確認瀏覽器權限已開啟！");

            }

        };

 

        document.getElementById('stopRec').onclick = () => {

            mediaRecorder.stop();

            document.getElementById('startRec').disabled = false;

            document.getElementById('stopRec').disabled = true;

            mediaRecorder.onstop = () => {

                const audioBlob = new Blob(audioChunks, { type: 'audio/wav' });

                const audioUrl = URL.createObjectURL(audioBlob);

                document.getElementById('audioPlayback').innerHTML = `<audio controls src="${audioUrl}"></audio>`;

            };

        };

 

        loadQuestion();

    </script>
